main
array[434] data;


procedure cleardata()
var i;
{
  let i <- 0;
  while i < 434 do
    let data[i] <- i+1;
    let i <- i + 1;
  od;
};


procedure output()
var i;
{
  let i <- 0;
  while i < 434 do
    call outputnum(data[i]);
    let i <- i + 1;
    if i == 10 then
        let i <- 434-i;
    fi;
  od;
  call outputnewline();
};


{
  call cleardata;
  call output;
}.
